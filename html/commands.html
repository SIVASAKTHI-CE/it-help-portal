CrowdStrike Falcon

crowdstrike-commands.sh bash
# Install CrowdStrike Falcon Sensor sudo dpkg -i Ubuntu_7.32.18504.deb sudo apt-get install -f # Set Customer ID (CID) sudo /opt/CrowdStrike/falconctl -s --cid=AD51434F6C8D4F60BE8051F85FB1F39C-A2 # Verify CID was set correctly sudo /opt/CrowdStrike/falconctl -g --cid # Enable + Start service sudo systemctl enable falcon-sensor sudo systemctl start falcon-sensor # Check service status sudo systemctl status falcon-sensor # Verify cloud registration (AID = registered) sudo /opt/CrowdStrike/falconctl -g --aid # Restart sensor sudo systemctl restart falcon-sensor # Uninstall sensor sudo apt-get remove falcon-sensor

Zscaler

zscaler-commands.sh bash
# Install Zscaler client sudo dpkg -i zscaler-client_*.deb sudo apt-get install -f # Service controls sudo systemctl status zscaler sudo systemctl restart zscaler sudo systemctl stop zscaler # Wayland display fix (if GUI doesn't load) sudo sed -i 's/#\?\WaylandEnable=false/WaylandEnable=false/' /etc/gdm3/custom.conf sudo systemctl restart gdm3 # Check Zscaler logs sudo journalctl -u zscaler -f

SSH / Network

ssh-network.sh bash
# Install OpenSSH server sudo apt install openssh-server openssh-client -y # SSH service controls sudo systemctl status ssh sudo systemctl enable ssh sudo systemctl start ssh # Allow SSH through firewall sudo ufw allow ssh sudo ufw status # Connect to remote machine ssh username@192.168.x.x # Check IP address ip addr show hostname -I # Ping test ping -c 4 8.8.8.8

Docker

docker-commands.sh bash
# Start all services docker compose up -d # Stop all services docker compose down # View running containers docker ps # View logs (live) docker logs it-help-portal -f docker logs it-ai-backend -f # Rebuild and restart docker compose up -d --build --force-recreate # Remove all stopped containers docker system prune -f # Check disk usage docker system df

Ruby on Rails

rails-commands.sh bash
# Install RVM + Ruby curl -sSL https://get.rvm.io | bash -s stable rvm install 3.2.2 rvm use 3.2.2 # Install Rails gem install rails -v 7.0 # MySQL sudo apt install mysql-server -y sudo mysql -u root ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY ''; # New Rails app rails new myapp -d mysql cd myapp rails db:create rails server

System

system-commands.sh bash
# Update system sudo apt update && sudo apt upgrade -y # Check disk space df -h # Check RAM usage free -h # Check running processes htop # Check open ports sudo netstat -tulpn sudo ss -tulpn # Check system logs sudo journalctl -xe # Restart system sudo reboot # Shutdown sudo shutdown -h now
